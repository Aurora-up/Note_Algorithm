```c++
#include<iostream>
using namespace std;
const int N = 10;
int n;
int path[N];   // 储存我们当前的状态
bool st[N];    // 用来记录我们用过的点

void dfs(int u){
    // 走到了第 n 个位置，并且在 n 这个状态下我们已经将三个位置填满，将其输出即可
    if(u > n){
        for(int i = 1;i <= n;i++) printf("%d ",path[i]);
        puts("");
        return ;
    }
    //  u < n ,我们就开始枚举寻找
    for(int i = 1 ;i <= n;i++){   
        // 当前点为遍历过 时，我么们就可以使用该点
        if(!st[i]){
            path[u] = i;  // 将用过的 i 放到我们当前状态里面去，即插入到我们的剩余位置当中 
            st[i] = true;   // 然后记录该店已经被用过
            dfs(u+1);     // 继续再递归到下一个点
            
            // 恢复现场
            // path[i] = 0;  没有必要做，因为每一次都会有 path[i] 被覆盖
            st[i] = false;
        }
    }
}

int main(){
    cin >> n;
    dfs(1);   //  从第 0 位置开始向下走
    return 0;
}
```

```c++
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;

typedef pair<int,int> PII;

const int N = 110;

int n,m;
int g[N][N]; // 存储 图
int d[N][N]; // 存储 每个点 到 起点的距离

PII q[N*N],Prev[N][N]; // 队列 和 路径存储

int bfs(){
    // 初始化队列
    int hh = 0,tt=0;
    q[0]={0,0};
    
    // 初始化距离
    memset(d,-1,sizeof d);  //  距离 -1 表示没有走国过
    d[0][0] = 0;  // 表示 走过
    
    // 用向量来表示 上下左右 移动
    int dx[4] = {-1,0,1,0}, dy[4] = {0,1,0,-1};
    
    while(hh<=tt){
        auto t = q[hh++];  // 取出对头
        
        // 枚举 4个方向
        for(int i=0;i<4;i++){
            // 上下左右 任一个方向的下一点
            int x = t.first + dx[i] , y = t.second + dy[i];
            
            // 边界内，是可以走的 图中的点，并且 这个点没有走过（距离 = -1）
            if(x >= 0 && y >= 0 && y < m && g[x][y] == 0 && d[x][y] == -1){ // 第一次搜到时
                
                d[x][y] = d[t.first][t.second] + 1; // 记录走过
                
                Prev[x][y] = t;
                
                q[++tt] = {x,y};
            }
        }
    }
    
    //  输出走过的路径
    int x = n - 1,y = m - 1;
    while(x || y){
        cout << x <<' '<< y << endl;
        auto t = Prev[x][y];
        x = t.first,y = t.second;
    }
    
    return d[n-1][m-1];
}


int main(){
    cin>>n>>m;
    
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
            scanf("%d",&g[i][j]);
    
    cout<< bfs() << endl;
    return 0;
}
```

#### 最小步数  845：八数码

![](image/acw_845.png)

```
主要问题：如何状态表示？  (可以使用 string类型)   使用 queue<string>
    	如何记录每个状态的距离？unordered_map<string.int>
		如何进行状态转移？ (可以是 先从 string --> 3 * 3 的矩阵 --> string)
```

```c++
#include<bits/stdc++.h>
using namespace std;

int bfs(string start){
    //  定义末状态
    string end = "12345678x";
    
    queue<string> q;
    unordered_map<string,int> d;  //  使用哈希表来存储距离
    
    // 初始化队列和距离
    q.push(start);
    d[start] = 0;
    
    //  使用方向数组来进行移动
    int dx[4] = {-1, 0 , 1 ,0};
    int dy[4] = {0 , 1 , 0 ,-1};
    
    while(q.size()){
        // 每次循环的队头替换
        auto t = q.front();
        q.pop();
    
        int distance = d[t];
        
        if(t == end) return distance;
        
        //  状态转移：  
        int k = t.find('x'); //  找到 x 的位置；
        
        int x = k / 3, y = k % 3;  //  将一维坐标转化为 3 * 3 矩阵坐标
        
        for(int i = 0 ; i < 4  ; i++){
            int a = x + dx[i];
            int b = y + dy[i];
            
            if(a >= 0 && a < 3 && b >= 0 && b < 3){
                
                // 将移动后的状态交换，注意将 3 * 3 矩阵的坐标转换为一维坐标
                swap(t[k],t[a * 3 + b]); // 状态更新
                
                if(!d.count(t)){
                    
                    d[t] = distance + 1;
                    q.push(t);

                }
                
                swap(t[k],t[a * 3 + b]); // 恢复状态
                
            }
        }
    }
    return -1;
}


int main()
{
    string start;
    for(int i = 0 ; i < 9 ; i++){
        char c;
        cin >> c;
        start +=c ;
    }
    cout << bfs(start) << endl;
    return 0;
};
```

```c++
// 基本写法
#include<bits/stdc++.h>
using namespace std;
const int N = 100010, M = N * 2;

int h[N],e[N],ne[N],idx;  
bool st[N]; //  深搜和宽搜的每个节点都只会遍历一次，所以我们需要 记录某个点 是否被搜过

void add(int a,int b){  
    e[idx] = b,ne[idx] = h[a],h[a] = idx++;
}

void dfs(int u){  // u  表示当前搜到的 点
    st[u] = true;  // 标记搜到的 点
    
    //  遍历一下 u 的所有  出边
    for(int i = h[u] ; i != -1 ; i = ne[i]){
        // 存一下 出边
        int j = e[i];
        if(!st[j])  //  判断这些点有没有被搜过
        {
            dfs(j);
            cout << j << endl;  //  输出 dfs 路径，是倒序的（栈）
        }
    }
}

    while(m--)
    {
        int a, b;

    }
```

##### 使用邻接表来存储图

```
将每个点 分离 后剩余连通块的最大值  的 最小值 作为树的重心
    在深度优先遍历过程中就可以计算出每个点的分离后的连通块节点的个数   时间复杂度 o(n+m)
    
    eg: 4 的子节点的个数可以分别算出，再通过所有的节点的个数 n - size_4 的就可以算出父节点的个数
        那么其连通块中节点的个数也就算出了
```

```
 dfs 参数和返回值的设计：
 	1：首先的需求是遍历整个树。
 	2：在遍历的同时求得 去除重心之后 连通块中节点数量的最大值。
 		在实现 2 的时候可以在 枚举的时候将将枚举到的当前点删掉，并求 此时已有连通块的max
 		同时对 max 求 min 即可。
 
int dfs(int u) //  从 u 号节点开始搜，返回的是以 u 为根的子树的节点个数。 
```

```c++
#include<bits/stdc++.h>
using namespace std;

const int N = 100010,M = N * 2;
int n;
int h[N],e[M],ne[M],idx;
bool st[N];

int ans_min = N;

void add(int a,int b){
    e[idx] = b, ne[idx] = h[a],h[a] = idx++;
}


// 设  u  为根的子树中点的数量 , 返回的是以 u 为根的子树中节点个数。
int dfs(int u){
    st[u] = true;
    
    //  sum：当前子树的大小（从 1 开始）
    //  res：存储将 u 删除之后所剩连通块 节点个数的最大值。
   	int sum = 1,res = 0;
    
    for(int i = h[u] ; i != -1 ; i = ne[i]){
        int j = e[i];
         // 搜 左右 子节点
        if(!st[j]) {
            int s = dfs(j);// 暂存 左右子节点的数量
        	res = max(res,s);
            //s 是以 u 的儿子 j为根的子树中的节点个数，所以也是u 子树节点个数的一部分
            sum += s;  
        }
    }
    
    // n - sum 就是将当前节点删掉之后树上的节点个数 （也是一个连通块）
    res = max(res , n - sum); 
    
    // 求 max 中的 min
    ans_min = min(res,ans_min);
        
    return sum;
}

int main(){
    cin >> n 
    memset(h,-1,sizeof h);
    
    for(int i = 0;i < n-1;i++){
        int a,b;
        cin >> a >> b;
        add(a,b),add(b,a);
        
    }    
    dfs(1);
    cout << ans_min << endl;
    return 0;
}
```

#### 宽度优先遍历

![](image/bfs_BianLi.png)

树与图的广度优先遍历需要使用一个队列来实现。起初，队列中仅包含一个起点（eg: 1 号节点）。在广度优先遍历中，需要不断从队友取出一个节点 $x$ ，对于 $x$ 面对的多条分支，把沿着每支到达的下一个节点（如果尚未访问过）插入队尾。重复执行上述过程，直至队空。

![](image/BFS_0.png)

时间复杂度 $o(N + M)$ ， $N$ 为点数，$M$ 为边数。

**注意：普通的 BFS 是在入队时判重。**

```c++
int d[N]; // 即是距离数组，又可以用来判断该点是否被用过。
void bfs(){
    int hh = 0,tt = 0;
    q[0] = 1;
    
    memset(d,-1,sizeof d);
    d[1] = 0;  //  将起始状态添加到队列中
    
    while(hh <= tt){ //  队列不空时
        //  存队头
        int t = q[hh++];
        //  遍历 t 的所有邻节点
        for(int i=h[t];i != -1;i=e[i]){
            int j = e[i];
            //  如果 未被遍历
            if(d[j] == -1){
                d[j] = d[t] + 1; //  更新距离
                q[++t] = j;  //  入队
                
                cout << j << endl;
            }
        }
    }	
}
```



代码中 `d[N]` 数组，对于一个树来说，`d[x]` 就是 `x` 在树中的深度。

对于一张图来说（**边权均为** $1$），`d[x] ` 是点 `X` 的层次（从起点 $1$ 走到点 $x$ 需要经过的最少点数），也即**最短路**。也即解决的是 **初始状态 到  每个状态 最小步数的问题。**

从遍历的顺序来看，广搜是按照层序的遍历方式。具有下面**重要性质**。

**1：两段性**：在访问完所有第 $i$ 层节点之后，才会开始访问第 $i + 1$ 层节点。

**2：单调性**：任意时刻，队列中至少有两个层次的节点。若其中一部分节点属于第 $i$ 层，则另一部分节点属于第 $i + 1$ 层，并且所有第 $i$ 层节点排在第  $i + 1$ 层节点之前。

#### 朴素Dijstar 算法

**算法流程：**

1：初始化 `dist[1] = 0` ,其余节点正无穷大。

2：找出一个未标记的，`dist[t]` 最小的节点 `t` ，然后标记节点 `t`。

3： 扫描节点 `x` 的所有出 `(x ,y, z)` ，若  `dist[y] > dist[x] + z`,则使用 `dist[x] + z` 更新 `dist[y]`.

4：然后重复 2~3 步骤，直到所有节点被标记。

**注意：** Dijkstra 算法中时是在**节点出队的时标记**。即：其 到起点的最短路已经被确定。

```c++
bool st[N]; //  标记数组，用于该点已经被用过去更新最短路，当前点到起点的最短路已确定。
int dist[N];//  最短路数组

void Dijkstra()
{
    //  初始化距离    
    memset(dist,0x3f,sizeof dist);
    memset(st, 0 , sizeof st);
    //  初始化最短路的起点的距离
    dist[1] = 0; 
    
    // 遍历寻找 n - 1 次
    for(int i = 1 ; i < n ; i++){
        int x = -1;
        
        //  这里遍历的目的是要找到未标记 节点中 dist 最小的
        for(int j = 1 ; j <= n ; j++)
      // 未被使用去更新其他点 &&  (未在已更新的最短路上 || 当前距离大于可以被更新的距离)
            if(!st[j] && (t == -1 || dist[t] > dist[j])) 
                x = j; //  符合条件，替换
        
        if(x == n) break; // 如果找到，直接返回
        
        st[x] = true; // 当前点的最短路已确定。（已被用过去更新其他点）
        
        //  用全局最小值 去更新其他节点。
        for(int y = 1 ; y <= n ; y++)
            // 用 1 到 t 距离 + t到j的这条边来更新到 j 最短路
            dist[y] = min(dist[y],dist[x] + g[t][y]); 
    }
}

/*
1:关于 memset(dist, 0x3f,sizeof dist)
    memset 是按照字节来进行初始化， 我们这里的  int 是 4 个字节  
    所以初始化之后的  dist[N] 中的元素是  0x3f3f3f3f。

2:起点初始不能被标记。即 st[1] = 1 不写;

因为 标记数组 的作用就是当这个点的已经被用去跟新其他点后,该点到起点的最小值已经被确定,
但如果初始时就已经 st[1] = 1,那么就会导致不能进入循环更新最短路，从而不进行 Dijkstra的过程

上述疑惑的原因主要是 来源于 spfa 算法的  st[] 数组，spfa 算法中的 st[] 数组指的是
该点是否在队列中。

*/
```

```c++
 Dijkstra 算法是不断的找 dist 最小的点去更新其他点到起点的距离，然后标记这个点，再去找下个
```

```c++
#include<algorithm>
#include<cstring>
#include<queue>

const int N = 150010;

int h[N],e[N],ne[N],idx,w[N]; // 使用邻接表 来存储稀疏图

    e[idx] = b,w[idx] = c,ne[idx] = h[a],h[a] = idx++;
}

int dijkstra(){
    memset(dist,0x3f,sizeof dist);
    dist[1] = 0;
    
    // 使用优先队列来维护距离，小根堆
    priority_queue<PII,vector<PII>,greater<PII>> heap;   
    heap.push({0,1}); // 起点入队
    
    while(heap.size()){
        auto t = heap.top(); // 取出堆中 表示距离最小的点
        heap.pop(); 
        
        int ver = t.second,distance = t.first; // ver 是编号， distance 是距离
        if(st[ver]) continue;  // 说明这个点之前出现过，直接过
        
        st[ver] = true; // 优化操作，可以减少每次的判点数量
        
        for(int i = h[ver] ; i != -1 ; i = ne[i]){  // 再用当前的点来更新最短路
            int j = e[i];
            if(dist[j] > distance + w[i]){   // 如过当前距离大于之前的 距离就更新
                dist[j] = distance + w[i];
                heap.push({dist[j],j});   // 再将 j 这个点放到优先队列中
            }
        }
    }
    
    if(dist[n] == 0x3f3f3f3f) return -1;
    return dist[n];
}
```

#### Bellman-Ford 算法

```c++
for n 次            (k 次，就会实现不超过 k 条边的最短距离)
    进行一次备份，防止 使用上次迭代到的点 去更新最短距离
    for 所有边   , a  b  w  , a --w--> b   --->  可以使用结构体来存
    	dist[b] =  min(dist[b],dist[a] + w)   更新最短距离 （松弛操作） ，注意这里的dist[a]是之前拷贝的未更新的

循环完后 有：
        dist[b] <= dist[a] + w;   


可以用来找负环（时间复杂度很高，一般不用）
当循环 n 次之后，最短距离有 n 条边话，那么这条最短距离上就有 n+1 点，那么就肯定有环的存在


注意：负环 可能回造成最短路径不存在
    当 负环 不在最短路径所要经过的路径时就不会影响最短路（不能在负环中一直转，要限制）
    当我们限制最短路经过的边的时候，有负环也不会影响最短路
```

### SPFA 算法

给定一张有向图，若对于图中某一条边 （x, y, z） 即：从 x 到 y 有一条边权为 z 的路径。有                              `dist[y] <= dist[x] + z`  成立，则称该边满足三角不等式。

**若所有的边都满足三角不等式，则 `dist` 数组就是所求的最短路** 

```c++
在 ballman_ford 算法当中， for 循环每次都要遍历所有的边来更新,但是不是每一次都会更新成功；
    
    更新步骤中 dist[b] = min(dist[b],dist[a] + w); --> 其中 只有当 dist[a] 变小才能达到优化的目的；
    
    而 spfa 就是在这里优化，使用 广搜 来优化；
    队列中存放就是 dist 变小的节点，用它去更新后面的节点。
    即只有那些 dist 变小的节点更新其他节点才不会是 无效操作。
   
    
    1: 先将起点放到队列中； queue <- 1;
	2:  while(queue != null){
        
        //  queue 中存放所有变小的节点
        t = queue.top()  // 取出队头 t 
        queue.pop();  // 删掉队头
        
      // 所有与 t 有连接的点都有可能被被更新; 所以要 更新 t 的所有 出边； eg : t -w-> b
        dist[b] = min(dist[b],dist[t] + w);
        
        // 如果更新成功的话；就把 b 加入队列; 
        queue.push(b);

        // 思路：更新过 b ， 再用 b 来更新其他的; 因为这里的 t 就是已经变小的，所以它才有资格去更新其他的点
    }

// 这里 queue 也是并不局限，也可以使用 priority_queue , heap 等等：
```

#### 851:spfa 求最短路

<img src="image/acw_851.png" style="zoom:50%;" />

```c++
#include<iostream>
#include<queue>
#include<cstring>
using namespace std;

const int N = 100010;
int n , m;
int h[N],e[N],ne[N], w[N] ,idx ;
int dist[N];
bool st[N]; // 是判断 该点是否在队列中

void add(int a, int b,int c)
{
    e[idx] = b , w[idx] = c,ne[idx] = h[a],h[a] = idx++;
}

int spfa()
{
    memset(dist , 0x3f , sizeof dist);
    memset(st , 0 , sizeof st); // 初始所有点都不在队列中
    dist[1] = 0;
    
    queue<int> q;
    q.push(1);
    st[1] = true;
    
    while(q.size())
    {
        int t = q.front();
        q.pop();
        st[t] = false;  //  取出后记得 恢复判重数组。
        
        for(int i = h[t] ; i != -1 ; i = ne[i])
        {
            int j = e[i];
            if(dist[j] > dist[t] + w[i])
            {
                dist[j] = dist[t] + w[i];
                if(!st[j])
                {
                    q.push(j);
                    st[j] = true;
                }
            }
        }
        
    }
    
    return dist[n];  
}

int main()
{
    cin >> n >> m;
    
    memset(h, -1, sizeof h);
    
    while (m -- ){
        int a, b ,c;
        scanf("%d%d%d",&a,&b,&c);
        add(a,b,c);
    }
    
    int t = spfa();
    
    if(t == 0x3f3f3f3f) puts("impossible");
    else printf("%d",t);
    
    return 0;
}
```

#### spfa 算法求负环

```
求负环的时候需要 开一个记录每个边 从起点到这条边的路线数。
    int cnt[N];
当 cnt[t] >= n 的时候，由抽屉原理 ， cnt[t] = n 的时说明  1 - t 至少经过 n 条边，
    那么 1 - t 之间就会有 n + 1 个点；
    这就说明一定 有两个点的值是相同的，也就说明了存在负环。
    
    在 dist[j] = dist[t] + w[i]; 更新之后
       cnt[j] = cnt[t] + 1;
    
    

注意： 这里我们求负环的时候，这个 负环有可能不在 1 号点所在的最短路中，
    所以我们需要将所有点都加到队列中去。
    
```

##### 852:spfa 判断负环

https://www.acwing.com/problem/content/854/

<img src="image/acw_852.png" style="zoom:50%;" />

```c++
#include<iostream>
#include<queue>
#include<cstring>
using namespace std;

const int N = 100010;
int n , m;
int h[N],e[N],ne[N], w[N] ,idx ;
int dist[N], cnt[N];
bool st[N];

void add(int a, int b,int c)
{
    e[idx] = b , w[idx] = c,ne[idx] = h[a],h[a] = idx++;
}

bool spfa()
{  
    queue<int> q;
    
    for(int i = 1 ; i <= n ; i++)
    {
        st[i] = true;
        q.push(i);
    }
    
    while(q.size())
    {
        int t = q.front();
        q.pop();
        
        st[t] = false;
        
        for(int i = h[t] ; i != -1 ; i = ne[i])
        {
            int j = e[i];
            if(dist[j] > dist[t] + w[i])
            {
                dist[j] = dist[t] + w[i];
                cnt[j] = cnt[t] + 1;
                if(cnt[j] >= n) return true;
                if(!st[j])
                {
                    q.push(j);
                    st[j] = true;
                }
            }
        }
        
    }
    
    return false;  
}

int main()
{
    cin >> n >> m;
    
    memset(h, -1, sizeof h);
    
    while (m -- ){
        int a, b ,c;
        scanf("%d%d%d",&a,&b,&c);
        add(a,b,c);
    }

    if(spfa()) puts("Yes");
    else puts("No");
    
    return 0;
}
```

### Floyd 算法

```
用 Floyed 来求 多源汇最短路

void floyd()
{
    for(int k = 1 ; k <= n ; k++)
        for(int i = 1 ; i <= n ; i++)
            for(int j = 1 ; j <= n ; j++)
                d[i][j] = min(d[i][j],d[i][k] + d[k][j]);
    
}

Floyd本质上是一个动态规划的思想，每一次循环更新经过前k个节点，i到j的最短路径。

不需要特意存图，因为dist数组本身就可以从邻接矩阵拓展而来。
初始化的时候，我们把每个点到自己的距离设为0，
每新增一条边，就把从这条边的起点到终点的距离设为此边的边权（类似于邻接矩阵）。其他距离初始
化为INF（一个超过边权数据范围的大整数，注意防止溢出）····

```

#### 854:Floyd 算法

```
floyd 本身是个动态规划算法，在代码实现的时候省去了一维状态。
原状态是：f[i, j, k]表示从i走到j的路径上除了 i, j 以外不包含点k的所有路径的最短距离。

那么f[i, j, k] = min(f[i, j, k - 1), f[i, k, k - 1] + f[k, j, k - 1]。

因此在计算第 k 层的f[i, j]的时候必须先将第 k - 1 层的所有状态计算出来，
所以需要把k放在最外层。

```

```
这里直接 计算了最小值。
传递闭包问题中的  当 i 可达 k , 并且 k 可达 j 的情况下 再更新 d[i][j];
就可以 最终直接判断 d[i][j] 是否等于 INF 来判断 可达，即有无最小值。
```

```c++
#include<iostream>
using namespace std;
const int INF = 0x3f3f3f3f;
const int N = 210;

int n, m ,Q;

int d[N][N]; //  floyd() 之后的就是 a ,b 之间的最短距离

void floyd()
{
    for(int k = 1 ; k <= n ; k++)
        for(int i = 1 ; i <= n ; i++)
            for(int j = 1 ; j <= n ; j++)
                d[i][j] = min(d[i][j],d[i][k] + d[k][j]);
    
}

int main()
{
    scanf("%d%d%d", &n, &m,&Q);
    
    for(int i = 1 ; i <= n ; i++)
        for(int j = 1 ; j <= n ; j++)
            if(i == j) d[i][j] = 0;
            else d[i][j] = INF;
            
            
    while (m -- )
    {
        int a, b, c;
        scanf("%d%d%d",&a,&b,&c);
        
        d[a][b] = min(d[a][b], c);
    }
    
    floyd();
    
    while(Q--)
    {
        int a, b;
        scanf("%d%d", &a, &b);
        if(d[a][b] > INF / 2) puts("impossible"); //  可能存出负权边
        else printf("%d\n",d[a][b]);
    }
    
    
    return 0;
}
```
