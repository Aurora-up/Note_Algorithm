```c++
所有的路径合起来就是一棵树，但是我们存的只是每一次的路径，不需要整棵树存起来，并且我们的回溯不需

    回溯的注意点：(恢复现场)
   		 一定要回的是我们之前的样子，就是我们下去的时候什么样，回来的时候就是什么样
```

```c++
#include<iostream>
using namespace std;
const int N = 10;
int n;
int path[N];   // 储存我们当前的状态
bool st[N];    // 用来记录我们用过的点

void dfs(int u){
    // 走到了第 n 个位置，并且在 n 这个状态下我们已经将三个位置填满，将其输出即可
    if(u > n){
        for(int i = 1;i <= n;i++) printf("%d ",path[i]);
        puts("");
        return ;
    }
    //  u < n ,我们就开始枚举寻找
    for(int i = 1 ;i <= n;i++){   
        // 当前点为遍历过 时，我么们就可以使用该点
        if(!st[i]){
            path[u] = i;  // 将用过的 i 放到我们当前状态里面去，即插入到我们的剩余位置当中 
            st[i] = true;   // 然后记录该店已经被用过
            dfs(u+1);     // 继续再递归到下一个点
            
            // 恢复现场
            // path[i] = 0;  没有必要做，因为每一次都会有 path[i] 被覆盖
            st[i] = false;
        }
    }
}

int main(){
    cin >> n;
    dfs(1);   //  从第 0 位置开始向下走
    return 0;
}
```

```c++
PII q[N*N],Prev[N][N]; // 队列 和 路径存储

int bfs(){
    // 初始化队列
    int hh = 0,tt=0;
    q[0]={0,0};
    
    // 初始化距离
    memset(d,-1,sizeof d);  //  距离 -1 表示没有走国过
    d[0][0] = 0;  // 表示 走过
    
    // 用向量来表示 上下左右 移动
    int dx[4] = {-1,0,1,0}, dy[4] = {0,1,0,-1};
    
    while(hh<=tt){
        auto t = q[hh++];  // 取出对头
        
        // 枚举 4个方向
        for(int i=0;i<4;i++){
            // 上下左右 任一个方向的下一点
            int x = t.first + dx[i],y = t.second + dy[i];
            
            // 边界内，是可以走的 图中的点，并且 这个点没有走过（距离 = -1）
            if(x >= 0 && y >= 0 && y < m && g[x][y] == 0 && d[x][y] == -1){ // 第一次搜到时
                
                d[x][y] = d[t.first][t.second] + 1; // 记录走过
                
                Prev[x][y] = t;
                
                q[++tt] = {x,y};
            }
        }
    }
    
    //  输出走过的路径
    int x = n-1,y = m-1;
    while(x || y){
        cout<<x<<' '<< y<<endl;
        auto t = Prev[x][y];
        x = t.first,y = t.second;
    }
    
    
    
    return d[n-1][m-1];
}
```

```c++
//  acw845
主要问题：如何状态表示？  (可以使用 string类型)   使用 queue<string>
    	如何记录每个状态的距离？unordered_map<string.int>
		如何进行状态转移？ (可以是 先从 string --> 3 * 3 的矩阵 --> string)
    

#include<iostream>
#include<cstring>
#include<algorithm>
#include<unordered_map>
#include<queue>

using namespace std;

int bfs(string start){
    //  定义末状态
    string end = "12345678x";
    
    queue<string> q;
    unordered_map<string,int> d;  //  使用哈希表来存储距离
    
    // 初始化队列和距离
    q.push(start);
    d[start] = 0;
    
    //  使用方向数组来进行移动
    int dx[4] = {-1, 0 , 1 ,0};
    int dy[4] = {0 , 1 , 0 ,-1};
    
    while(q.size()){
        // 每次循环的队头替换
        auto t = q.front();
        q.pop();
    
        int distance = d[t];
        
        if(t == end) return distance;
        
        //  状态转移：  
        int k = t.find('x'); //  找到 x 的位置；
        
        int x = k / 3, y = k % 3;  //  将一维坐标转化为 3 * 3 矩阵坐标
        
        for(int i = 0 ; i < 4  ; i++){
            int a = x + dx[i];
            int b = y + dy[i];
            
            if(a >= 0 && a < 3 && b >= 0 && b < 3){
                
                // 将移动后的状态交换，注意将 3 * 3 矩阵的坐标转换为一维坐标
                swap(t[k],t[a * 3 + b]); // 状态更新
                
                if(!d.count(t)){
                    
                    d[t] = distance + 1;
                    q.push(t);

                }
                
                swap(t[k],t[a * 3 + b]); // 恢复状态
                
            }
        }
    }
    return -1;
}

    for(int i = 0 ; i < 9 ; i++){
```

#### 朴素Dijstar 算法

```c++
Dijkstra 算法：用于计算一个 节点到其他所有节点的最短路。  o(n^2)


算法思路：
    1： 初始化距离， dist[1] = 0  dist[i] = +∞(具体实现于很大的数)
    
    2： for(i -> (1,n))     				s[N] --> 存放当前已确定最短距离的点
    	 t <- 不在 s[N] 中的距离最近的点
         将 t 放入 s[N] 中
         用 t 更新其他点的距离  <--  从 t 出去的所有边组成的路径能不能更新其他点到 1 的距离

        
   	  循环 n 次之后就可以求出 n个点 的最短的距离，也就是每个点到起点的最短距离
  
        
```

```c++
// acw 849
#include<iostream>
#include<algorithm>
#include<cstring>

using namespace std;

const int N = 510;

int n,m;
int g[N][N]; // 使用邻接矩阵来存储 稠密图
int dist[N]; // 当前的最短距离
bool st[N];  // 表示当前点的最短路是否已经确定

int dijkstra(){
    // 初始化距离
    memset(dist,0x3f,sizeof dist);
    dist[1] = 0;
    
    // 遍历寻找
    for(int i = 0 ; i < n ; i++){
        int t = -1;
        //  遍历寻找最短路
        for(int j = 1 ; j <= n ; j++)
            // 最短路未确定&&当前的路不是最短的
            if(!st[j] && (t == -1 || dist[t] > dist[j])) 
                t = j;
        
        if(t == n) break; // 如果找到，直接返回
        
        st[t] = true;
        
        for(int j = 1 ; j <= n ; j++)
            // 用 1 到 t 距离 + t到j的这条边来更新到 j 最短路
            dist[j] = min(dist[j],dist[t] + g[t][j]); 
    }
    
    if(dist[n] == 0x3f3f3f3f) return -1;  // 最短路不存在
    return dist[n];
}

int main(){
    cin >> n >> m;
    
    memset(g,0x3f,sizeof g);
    
    while(m--){
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        g[a][b] = min(g[a][b],c);  // 处理重边和自环，只保留长度最短的那条边
    }
    
    int t = dijkstra();
    
    printf("%d\n",t);
    return 0;
}
```

#### 堆优化版的 Dijstar 算法

```c++
o(mlogn)
图中节点增多时,朴素的 Dijkstra 算法的 o(n^2) 会爆掉。
    所以使用 堆来存储每个节点到 第一个点的最小值，
    每次我们都将 堆顶 t 输出，堆顶是 1 到 t 的最小距离，不必在  1 到 t 的所有距离中
    加上    t 到 下一个点 的距离  后求最小值，直接以最小 1 - t 来加。
    
    使用小根堆，将最小的数放在堆顶，使得这一步骤的时间复杂度 由 o(n^2) -> o(n)
	使用堆后，在后续的 更新最短路的距离过程中，因为我们要在 推中查到这个节点并修改其最短距离
    时间复杂度是 o(mlogn)
```



```c++
#include<iostream>
#include<algorithm>
#include<cstring>
#include<queue>

using namespace std;

typedef pair<int,int> PII;

const int N = 150010; 

int n,m;
int h[N],e[N],ne[N],idx,w[N]; // 使用邻接表 来存储稀疏图
int dist[N];
bool st[N];

void add(int a,int b,int c){
    e[idx] = b,w[idx] = c,ne[idx] = h[a],h[a] = idx++;
}

int dijkstra(){
    memset(dist,0x3f,sizeof dist);
    dist[1] = 0;
    
    // 使用优先队列来维护距离，小根堆
    priority_queue<PII,vector<PII>,greater<PII>> heap;   
    heap.push({0,1}); // 加载第一个点的距离
    
    while(heap.size()){
        auto t = heap.top(); // 取出堆中 表示距离最小的点
        heap.pop(); 
        
        int ver = t.second,distance = t.first; // ver 是编号， distance 是距离
        if(st[ver]) continue;  // 说明这个点之前出现过，直接过
        
        st[ver] = true; // 优化操作，可以减少每次的判点数量
        
        for(int i = h[ver] ; i != -1 ; i = ne[i]){  // 再用当前的点来更新最短路
            int j = e[i];
            if(dist[j] > distance + w[i]){   // 如过当前距离大于之前的 距离就更新
                dist[j] = distance + w[i];
                heap.push({dist[j],j});   // 再将 j 这个点放到优先队列中
            }
        }
    }
    
    if(dist[n] == 0x3f3f3f3f) return -1;
    return dist[n];
}

int main(){
    cin >> n >> m;
    
    memset(h,-1,sizeof h);
    
    while(m--){
        int a,b,c;
        scanf("%d%d%d",&a,&b,&c);
        add(a,b,c);
    }
    
    int t = dijkstra();
    
    printf("%d\n",t);
    return 0;
}
```

#### Bellman-Ford 算法

```c++
for n 次            (k 次，就会实现不超过 k 条边的最短距离)
    进行一次备份，防止 使用上次迭代到的点 去更新最短距离
    for 所有边   , a  b  w  , a --w--> b   --->  可以使用结构体来存
    	dist[b] =  min(dist[b],dist[a] + w)   更新最短距离 （松弛操作） ，注意这里的dist[a]是之前拷贝的未更新的

循环完后 有：
        dist[b] <= dist[a] + w;   


可以用来找负环（时间复杂度很高，一般不用）
当循环 n 次之后，最短距离有 n 条边话，那么这条最短距离上就有 n+1 点，那么就肯定有环的存在


注意：负环 可能回造成最短路径不存在
    当 负环 不在最短路径所要经过的路径时就不会影响最短路（不能在负环中一直转，要限制）
    当我们限制最短路经过的边的时候，有负环也不会影响最短路
    
```

### SPFA 算法

```c++
在 ballman_ford 算法当中， for 循环每次都要遍历所有的边来更新,但是不是每一次都会更新成功；
    
    更新步骤中 dist[b] = min(dist[b],dist[a] + w); --> 其中 只有当 dist[a] 变小才能达到优化的目的；
    
    而 spfa 就是在这里优化，使用 宽搜 来优化；
    队列中存放就是 dist 变小的节点，用它去更新后面的节点。
    即只有那些 dist 变小的节点更新其他节点才不会是 无效操作。
    
    1: 先将起点放到队列中； queue <- 1;
	2:  while(queue != null){
        
        //  queue 中存放所有变小的节点
        t = queue.top()  // 取出队头 t 
        queue.pop();  // 删掉队头
        
        // 所有与 t 有连接的点都有可能被被更新; 所以要 更新 t 的所有 出边； eg : t -w-> b
        dist[b] = min(dist[b],dist[t] + w);
        
        // 如果更新成功的话；就把 b 加入队列; 
        queue.push(b);  
        
        // (加入之前还需判断 b 是否已经在 队列中)
        // 思路：更新过 b ， 再用 b 来更新其他的; 因为这里的 t 就是已经变小的，所以它才有资格去更新其他的点
    }
// 这里 queue 也是并不局限，也可以使用 priority_queue , heap 等等：

```

#### spfa 算法求负环

```
求负环的时候需要 开一个记录每个边 从起点到这条边的路线数。
    int cnt[N];
当 cnt[t] >= n 的时候，由抽屉原理 ， cnt[t] = n 的时说明  1 - t 至少经过 n 条边，
    那么 1 - t 之间就会有 n + 1 个点；
    这就说明一定 有两个点的值是相同的，也就说明了存在负环。
    
    在 dist[j] = dist[t] + w[i]; 更新之后
       cnt[j] = cnt[t] + 1;
    
    

注意： 这里我们求负环的时候，这个 负环有可能不在 1 号点所在的最短路中，
    所以我们需要将所有点都加到队列中去。
    
```

### Floyd 算法

```
用 Floyed 来求 多源汇最短路

void floyd()
{
    for(int k = 1 ; k <= n ; k++)
        for(int i = 1 ; i <= n ; i++)
            for(int j = 1 ; j <= n ; j++)
                d[i][j] = min(d[i][j],d[i][k] + d[k][j]);
    
}

Floyd本质上是一个动态规划的思想，每一次循环更新经过前k个节点，i到j的最短路径。

不需要特意存图，因为dist数组本身就可以从邻接矩阵拓展而来。初始化的时候，我们把每个点到自己的距离
设为0，每新增一条边，就把从这条边的起点到终点的距离设为此边的边权（类似于邻接矩阵）。其他距离初始
化为INF（一个超过边权数据范围的大整数，注意防止溢出）。
```

[Floyd 算法]:https://www.acwing.com/problem/content/856/

```c++
#include<iostream>
using namespace std;
const int INF = 0x3f3f3f3f;
const int N = 210;

int n, m ,Q;

int d[N][N]; //  floyd() 之后的就是 a ,b 之间的最短距离

void floyd()
{
    for(int k = 1 ; k <= n ; k++)
        for(int i = 1 ; i <= n ; i++)
            for(int j = 1 ; j <= n ; j++)
                d[i][j] = min(d[i][j],d[i][k] + d[k][j]);
    
}

int main()
{
    scanf("%d%d%d", &n, &m,&Q);
    
    for(int i = 1 ; i <= n ; i++)
        for(int j = 1 ; j <= n ; j++)
            if(i == j) d[i][j] = 0;
            else d[i][j] = INF;
            
            
    while (m -- )
    {
```
