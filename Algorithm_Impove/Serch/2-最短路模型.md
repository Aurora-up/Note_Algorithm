#### 1076：迷宫问题

https://www.acwing.com/problem/content/1078/

求解最短路的 路径打印。

```
通过将 判重 数组 bool st[N][N]  转换为  PII pre[N][N] 
对最短路的 路径进行标记。

1：图外不扩展
2：墙壁不扩展
3：已扩展过的点不扩展
```

```c++
#include<iostream>
#include<cstring>
#include<algorithm>

#define x first
#define y second

using namespace std;
typedef pair<int, int> PII;
const int N = 1010 ,M = N * N;

int n;
int g[N][N];    
PII q[M];
PII pre[N][N];   //  记录路径数组
int dx[4] = {-1 , 0 , 1 , 0} , dy[4] = {0 , 1 , 0 , -1}; // 偏移量

void bfs(int sx, int sy)
{
    int hh = 0 , tt = 0;
    q[0] = {sx , sy};
    
    memset(pre , -1 , sizeof pre);
    pre[sx][sy] = {0,0};
    
    while(hh <= tt)
    {
        PII t = q[hh ++];
        
        for(int i = 0 ; i < 4 ; i++)
        {
            int a = t.x + dx[i], b = t.y + dy[i];
            
            if(a < 0 || a >= n || b < 0 || b >= n) continue;
            if(g[a][b]) continue;           // 墙壁
            if(pre[a][b].x != -1) continue; // 该路没有走过
            
            q[++tt] = {a ,b};
            pre[a][b] = t;
        }
    }
}

int main()
{
    scanf("%d", &n);
    
    for(int i = 0 ; i < n ; i++)
        for(int j = 0 ; j < n ; j++)
            scanf("%d",&g[i][j]);
            
    bfs(n - 1,n - 1); // 从终点开始搜索，方便打印路径
    
    PII end(0 , 0);
    
    while(true)
    {
        printf("%d %d\n",end.x ,end.y);
        if(end.x == n - 1 && end.y == n - 1) break;
        end = pre[end.x][end.y];
    }
    
    return 0;
}
```

#### 180：武士风度的牛

https://www.acwing.com/problem/content/190/

```
本题 的扩展方式 不同， 是按照 日 字走的 8 个方向进行扩展。

注意： 偏移量 使用 纵轴为 X 轴，横轴 为 y
int dx[8] = {-2 , -1, 1, 2,  2,  1, -1, -2};
int dy[8] = { 1 ,  2, 2, 1, -1, -2, -2 ,-1};

```

```c++
#include <cstring>
#include <iostream>
#include <algorithm>

#define x first
#define y second

using namespace std;

typedef pair<int, int> PII;

const int N = 155, M = N * N;

int n, m;
char g[N][N];
PII q[M];

int dist[N][N];

int bfs()
{
    int dx[] = {-2, -1, 1, 2, 2, 1, -1, -2};
    int dy[] = {1, 2, 2, 1, -1, -2, -2, -1};

    int sx, sy;  //   遍历 图，找到 起点。
    for (int i = 0; i < n; i ++ )
        for (int j = 0; j < m; j ++ )
            if (g[i][j] == 'K')
                sx = i, sy = j;

    int hh = 0, tt = 0;
    q[0] = {sx, sy};

    memset(dist, -1, sizeof dist);
    dist[sx][sy] = 0;
	
    while (hh <= tt)
    {
        auto t = q[hh ++ ];

        for (int i = 0; i < 8; i ++ )
        {
            int a = t.x + dx[i], b = t.y + dy[i];
            
            if (a < 0 || a >= n || b < 0 || b >= m) continue;
            if (g[a][b] == '*') continue;
            if (dist[a][b] != -1) continue;
            if (g[a][b] == 'H') return dist[t.x][t.y] + 1;

            dist[a][b] = dist[t.x][t.y] + 1;
            q[ ++ tt] = {a, b};
        }
    }

    return -1;
}

int main()
{
    cin >> m >> n;

    for (int i = 0; i < n; i ++ ) cin >> g[i];

    cout << bfs() << endl;

    return 0;
}
```

#### 1100：抓住那头牛

https://www.acwing.com/problem/content/1102/

### **为什么在边权为 1 的时候，可以直接使用 bfs 来求解最短路问题：？**

```
1： 从 Dijkstar 的角度说明：

在 堆优化的 Dijkstar 算法当中， 对于 优先队列来说，我们 每次取得队头都是 边权值 最小的
节点， 而 当所有的边得权值 都为 1得时候，我们 直接使用 队列就可以，并且 由于存储关系，
存储的时候 是按照层级 进行存储的，所以 bfs 就会按照 层级 对图进行遍历。
	bfs 当中 普通队列的 队头，就像当中 堆优化Dijkstar 算法中的 优先队列（小根堆）的
的队头。
	
```

```
2：证明：
	bfs 当中的队列具有的性质
        1：两段性：最多有两段： 是由于 在我们维护的 队列中，我们只是用 hh 和 tt;
    并且每一次的扩展 都会将 这些点加在 队位，加上 本次扩展边的权值，就会一直保持单调性
        2：单调性：
	
	bfs 具有的性质 主要是由 按照 层级结构 对 节点 进行进行存储 
	[1,1,1, ... ,x,x,x,x, ... ,x + 1 ,x + 1 ,x + 1...]
    x 表示 该层点 到 根节点 的距离, 并且 边权值 为 1 , 所以下一层 距离 + 1 。
 所以 在 队列中的元素 满足单调性。
 
```



```
本题当中  人 向右走的方式 有 +1 , *2  操作。 向左走的方式 只有 -1 操作。
牛的位置 K 的范围是 10^5  ，因为有 *2 操作，所以当极限方式下， 数组需要开到  2 * 10^5。

```

```c++
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;

const int N = 2e5 + 10;
int n ,k;
int q[N];
int dist[N];

int bfs(){
    memset(dist , -1 ,sizeof dist);
    dist[n] = 0; //  人的 起点是 n
    q[0] = n;  //  队列起点也是 n
    
    int hh = 0 , tt = 0;
    
    while(hh <= tt)
    {
        int t = q[hh ++];
        // 首先判断 当前队头是否就是 k 位置。
        if(t == k)  return dist[k];
        
        if(t + 1 < N && dist[t + 1] == -1) //  t + 1 位置可以走
        {
            dist[t + 1] = dist[t] + 1; // 距离 
            q[++tt] = t + 1;  //  将 当前走到的 位置替换为 队头继续进行扩展。
        }
        if(t - 1 >= 0 && dist[t - 1] == -1) // t - 1位置可以走
        {
            dist[t - 1] = dist[t] + 1;
            q[++tt] = t - 1;
        }
        if(t * 2 < N && dist[t * 2] == -1) //  t * 2 位置可以走
        {
            dist[t * 2] = dist[t] + 1;
            q[++tt] = t * 2;
        }
    }
    
    return -1;
}

int main()
{
    cin >> n >> k;
    
    cout << bfs() << endl;

    return 0;
}
```
